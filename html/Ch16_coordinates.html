Section T16.

Context `{MT:Tarski_2D_euclidean}.
Context `{EqDec:EqDecidability Tpoint}.

Lemma exists_grid :
 exists O I J, ~ Col O I J /\ Per I O J /\ Cong O I O J.
Proof.
assert (T:=lower_dim).
destruct T as [O [I [X H]]].
exists O.
exists I.
assert (~ Col I O X) by (intro;unfold Col in *;intuition).
assert_diffs.
elim (ex_per_cong I O O X O I).
intros J. intros.
spliter.
exists J.
repeat split.
assert_diffs.
assert ( ~ Col J O I) by (apply (per_not_col J O I);auto).
Col.
Perp.
Cong.
auto.
auto.
Col.
auto.
Qed.

End T16.
