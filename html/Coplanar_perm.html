coplanar A B C D :=
  ∃ X, (Col A B X /\ Col C D X) \/ (Col A C X /\ Col B D X) \/ (Col A D X /\ Col B C X)


Lemma coplanar_perm_1 : ∀ A B C D,
  coplanar A B C D -> coplanar A B D C.
Proof.
intros A B C D HCop.
destruct HCop as [X H]; ∃ X.
induction H; try (induction H); spliter; Col5.
Qed.
