Class midpoint_theorem :=
{
  Point : Set;
  Line : Set;
  Segment : Set;
  UnorderedPair : Set;
  UnorderedTriple : Set;
  Quadrilateral4Uple : Set;
  Expr : Set;
  VExpr : Set;

  pset2 : Point -> Point -> UnorderedPair;
  pset2_prop : forall A B, pset2 A B = pset2 B A;
  pset3 : Point -> Point -> Point -> UnorderedTriple;
  pset3_prop1 : forall A B C, pset3 A B C = pset3 B C A;
  pset3_prop2 : forall A B C, pset3 A B C = pset3 B A C;
  pset4 : Point -> Point -> Point -> Point -> Quadrilateral4Uple;
  linePP : UnorderedPair -> Line;
  segPP : UnorderedPair -> Segment;
  
  zero : Expr;
  nvar :> nat -> Expr;
  dist : UnorderedPair -> Expr;

  dot : VExpr -> VExpr -> Expr;
  zerovect : VExpr;
  vect : Point -> Point -> VExpr;
  vplus : VExpr -> VExpr -> VExpr;
  vminus : VExpr -> VExpr -> VExpr;
  vuminus : VExpr -> VExpr;
  vmult : Expr -> VExpr -> VExpr;

  is_middle_segment : Point -> Segment -> Prop;

  is_parallelogram : Quadrilateral4Uple -> Prop;
  is_crossed : Quadrilateral4Uple -> Prop;
  are_parallel : Line -> Line -> Prop;
  are_collinear : UnorderedTriple -> Prop;
  is_symmetric : Point -> Point -> Point -> Prop;
  is_projected_on : Point -> Point -> Line -> Prop;

  middle_intro_sym : forall (A B C:Point),
  is_symmetric A B C ->
  is_middle_segment B (segPP (pset2 A C));

  same_dist_parallel_parallelogram_1 : forall (A B C D:Point),
  dist (pset2 A B) = dist (pset2 C D) ->
  are_parallel (linePP (pset2 A B)) (linePP (pset2 C D)) ->
  ~ is_crossed (pset4 A B C D) ->
  is_parallelogram (pset4 A B C D);

  same_dist_parallel_parallelogram_2 : forall (A B C D:Point),
  dist (pset2 A B) = dist (pset2 C D) ->
  are_parallel (linePP (pset2 A B)) (linePP (pset2 C D)) ->
  ~ is_crossed (pset4 B A C D) ->
  is_parallelogram (pset4 B A C D);

  same_middle_parallelogram : forall (A B C D M:Point),
  is_middle_segment M (segPP (pset2 A C)) ->
  is_middle_segment M (segPP (pset2 B D)) ->

  is_parallelogram (pset4 A B C D);

  same_middle_parallelogram_2 : forall (A B C D M:Point),
  is_middle_segment M (segPP (pset2 A C)) ->
  is_middle_segment M (segPP (pset2 B D)) ->

  is_parallelogram (pset4 B A C D);


  parallel_opp_lines_parallelogram_1 : forall (A B C D:Point),
  is_parallelogram (pset4 A B C D) ->
  are_parallel (linePP (pset2 A B)) (linePP (pset2 C D));


  parallel_opp_lines_parallelogram_2 : forall (A B C D:Point),
  is_parallelogram (pset4 A B C D) ->
  are_parallel (linePP (pset2 A D)) (linePP (pset2 B C));

  same_length_opp_parallelogram_1 : forall (A B C D:Point),
  is_parallelogram (pset4 A B C D) ->
  dist (pset2 A B) = dist (pset2 C D);

  same_length_opp_parallelogram_2 : forall (A B C D:Point),
  is_parallelogram (pset4 A B C D) ->
  dist (pset2 A D) = dist (pset2 B C);

  equal_vect_parallelogram : forall (A B C D:Point),
  is_parallelogram (pset4 A B C D) ->
  vect A B = vect D C;

  equal_vect_parallelogram_2 : forall (A B C D:Point),
  is_parallelogram (pset4 A B C D) ->
  vect B C = vect A D;



  middle_is_aligned : forall (A B M:Point),
  is_middle_segment M (segPP (pset2 A B)) ->
  are_collinear (pset3 A M B);



  middle_is_aligned_2 : forall (A B M:Point),
  is_middle_segment A (segPP (pset2 M B)) ->
  are_collinear (pset3 A M B);

  middle_is_aligned_3 : forall (A B M:Point),
  is_middle_segment B (segPP (pset2 M A)) ->
  are_collinear (pset3 A M B);

  middle_same_dist : forall (A B M:Point),
  is_middle_segment M (segPP (pset2 A B)) ->
  dist (pset2 A M) = dist (pset2 M B);



  middle_same_dist_2 : forall (A B M:Point),
  is_middle_segment M (segPP (pset2 A B)) ->
  dist (pset2 M A) = dist (pset2 M B);

  middle_same_dist_3 : forall (A B M:Point),
  is_middle_segment M (segPP (pset2 A B)) ->
  dist (pset2 A M) = dist (pset2 B M);

  middle_same_dist_4 : forall (A B M:Point),
  is_middle_segment M (segPP (pset2 A B)) ->
  dist (pset2 M A) = dist (pset2 B M);


  middle_double_vect : forall (A B M:Point),
  is_middle_segment M (segPP (pset2 A B)) ->
  vect A B = vmult (nvar 2) (vect A M);

  middle_double_vect_2 : forall (A B M:Point),
  is_middle_segment M (segPP (pset2 A B)) ->
  vect A B = vmult (nvar 2) (vect M B);

  middle_double_vect_3 : forall (A B M:Point),
  is_middle_segment M (segPP (pset2 A B)) ->
  vect B A = vmult (nvar 2) (vect M A);

  middle_double_vect_4 : forall (A B M:Point),
  is_middle_segment M (segPP (pset2 A B)) ->
  vect B A = vmult (nvar 2) (vect B M);



 parallel_sym : forall (A B C D:Point),
  are_parallel (linePP (pset2 A B)) (linePP (pset2 C D)) ->
  are_parallel (linePP (pset2 C D)) (linePP (pset2 A B));


 parallel_trans_1 : forall (A B C D E:Point),
  are_parallel (linePP (pset2 A B)) (linePP (pset2 C D)) ->
  are_collinear (pset3 C D E) ->
  are_parallel (linePP (pset2 A B)) (linePP (pset2 D E));

 parallel_trans_2 : forall (A B C D E:Point),
  are_parallel (linePP (pset2 A B)) (linePP (pset2 C D)) ->
  are_collinear (pset3 C D E) ->
  are_parallel (linePP (pset2 A B)) (linePP (pset2 C E));

 parallel_trans_3 : forall (A B C D E:Point),
  are_parallel (linePP (pset2 A B)) (linePP (pset2 C D)) ->
  are_collinear (pset3 A B E) ->
  are_parallel (linePP (pset2 B E)) (linePP (pset2 C D));

 parallel_trans_4 : forall (A B C D E:Point),
  are_parallel (linePP (pset2 A B)) (linePP (pset2 C D)) ->
  are_collinear (pset3 A B E) ->
  are_parallel (linePP (pset2 A E)) (linePP (pset2 C D));

 parallel_collinear : forall (A B C D:Point) (a:Expr),
  a <> zero ->
  vect A B = vmult a (vect C D) ->
  are_parallel (linePP (pset2 A B)) (linePP (pset2 C D));

 parallel_collinear_2 : forall (A B C D:Point) (a:Expr),
  vect A B = vect C D ->
  are_parallel (linePP (pset2 A B)) (linePP (pset2 C D));


 parallel_collinear_3 : forall (A B C D:Point) (a:Expr),
  vect A B = vmult a (vect C D) ->
  are_parallel (linePP (pset2 A B)) (linePP (pset2 C D))

}.

Require Import Euler_line.

Require Import aux.

Section T10.

Context `{M:Tarski_2D}.
Context `{EqDec:EqDecidability Tpoint}.

Definition pset2_impl (A B:Tpoint) := (A,B).

Definition psets_eq (ps1 ps2: Tpoint*Tpoint) :=
  match ps1,ps2 with
   (x1,y1),(x2,y2) => x1=x2 \/ y1=y2
  end.

Definition LineImpl := (Tpoint * Tpoint)%type. Definition linePPImpl (w:(Tpoint * Tpoint)%type) := w.
Definition SegmentImpl := (Tpoint * Tpoint)%type.
Definition segPPImpl (w:(Tpoint * Tpoint)%type) := w.

Definition are_parallel_impl L M :=
  match L,M with
   (A,B), (C,D) => Par A B C D
  end.

Lemma parallel_sym_impl : forall (A B C D:Tpoint),
  are_parallel_impl (linePPImpl (pset2_impl A B)) (linePPImpl (pset2_impl C D)) ->
  are_parallel_impl ((pset2_impl C D)) ( (pset2_impl A B)).
Proof.
intros.
unfold are_parallel_impl in *.
simpl in *.
Par.
Qed.

Definition pset3_impl (A B C:Tpoint) := (A,B,C).

Definition triple_eq (t1 t2: Tpoint * Tpoint * Tpoint) :=
 match t1,t2 with
  (X,Y,Z),(U,V,W) => (X=U /\ Y=V /\ Z=W) \/
                     (X=V /\ Y=U /\ Z=W)
 end.

Definition is_middle P s := match s with (X,Y) => is_midpoint P X Y end.

Definition are_collinear_impl s := match s with (X,Y,Z) => Col X Y Z end.

Lemma middle_is_aligned_impl : forall (A B M:Tpoint),
  is_middle M (segPPImpl (pset2_impl A B)) ->
  are_collinear_impl (pset3_impl A M B).
Proof.
intros.
simpl in *.
assert_cols.
Col.
Qed.

Lemma parallel_trans_1_impl : forall (A B C D E:Tpoint),
  D<>E ->
  are_parallel_impl (linePPImpl (pset2_impl A B)) (linePPImpl (pset2_impl C D)) ->
  are_collinear_impl (pset3_impl C D E) ->
  are_parallel_impl (linePPImpl (pset2_impl A B)) (linePPImpl (pset2_impl D E)).
Proof.
intros.
simpl in *.
apply par_col_par with C;Par;Col.
Qed.
