Lemma ex_unique_introduction :
   forall (A:Type)(P:A->Prop),
    (exists x, P x) ->
    (forall a b, P a -> P b -> a = b)->
    exists! x, P x.
Proof.
 intros.
 case (unique_existence P).
 destruct 1.
 split;auto.
 case H;intros x0 Hx0.
 exists x0;auto.
 case H1.
 split;auto.
Qed.

Theorem epsilon_ind :
   forall (A:Type)(i : inhabited A)(P:A->Prop)
          (Q:A->Prop), (exists x, P x)->
                       (forall x : A, P x -> Q x) ->
                       Q (epsilon i P).
Proof.
 intros A a P Q exP P_Q.
 apply P_Q.
  apply epsilon_spec; assumption.
Qed.

Theorem epsilon_equiv : forall (A:Type)(i : inhabited A)(P:A->Prop),
  (exists x, P x)<-> P (epsilon i P).
Proof.
 split.
 intros; apply epsilon_ind; auto.
 exists (epsilon i P); auto.
Qed.


Definition tau (A:Type)(i:inhabited A)(P:A->Prop) :=

Lemma forall_def : (forall P, P \/ ~ P) ->

      (epsilon i (fun (b':B) => DA a /\ R a b')).

Lemma dd'' : forall (A:Type)(B:A->Type)
                         (R:forall x:A, B x -> Prop),
      (forall x: A, inhabited (B x)) ->
      (sigT (fun f : forall x:A, B x =>
                     (forall x:A, (exists y:B x, R x y) -> R x (f x)))).
Proof.
 intros A B R i.
 exists (fun (x:A) => (epsilon (i x) (fun y => (R x y)))).
 intros x [y ry].
 apply (epsilon_ind). exists y; trivial.
 auto.
Qed.

       sigT (fun f : forall x:A, B x => (forall x:A, R x (f x))).

 intros.

 auto.
