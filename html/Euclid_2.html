Lemma playfair_s_postulate_implies_midpoints_postulate :
  playfair_s_postulate ->
  (∀ A B C P Q,
   ~ Col A B C ->
   is_midpoint P B C ->
   is_midpoint Q A C ->
   Par_strict A B Q P).
Proof.
intro HP; intros.
destruct (symmetric_point_construction P Q) as [P' HP'].
assert_diffs; assert_cols.
assert (~ Col A P C) by (intro; search_contradiction).
assert_diffs.
assert (HPara : Parallelogram_strict A P C P') by (apply mid_plgs with Q; finish).
assert (H' := playfair_s_postulate_implies_plgs_cong HP A P C P' HPara); spliter.
assert (Cong A P' B P) by (unfold is_midpoint in *; spliter; eCong).
assert (H' := playfair_s_postulate_implies_plgs_par_strict HP A P C P' HPara); spliter.
assert (Par A P' B P)
  by (apply par_col2_par with P C;
      try (apply par_strict_par); finish).
assert (HElim : Parallelogram A P' B P \/ Parallelogram A P' P B)
  by (apply playfair_s_postulate_implies_par_cong_plg_2; assumption).
induction HElim.

 destruct (playfair_s_postulate_implies_plg_mid HP A P' B P) as [M HM]; finish; spliter.
 treat_equalities.
 search_contradiction.

assert (H':= playfair_s_postulate_implies_plg_permut HP A P' P B).
destruct (playfair_s_postulate_implies_plg_par HP P' P B A);
try apply H'; finish; try intro; treat_equalities; Col.
assert (Par P Q A B)
  by (apply par_col_par_2 with P'; Col; Par; intro; treat_equalities; Col).
apply par_not_col_strict with P'; Col; Par.
 intro; destruct (not_strict_par A P' B P B); Col.
 assert_diffs; assert_cols; apply H; ColR.
Qed.
