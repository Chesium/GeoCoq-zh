decidability_of_intersection :=
  ∀ A B C D,
  (∃ I, Col I A B /\ Col I C D) \/
  ~ (∃ I, Col I A B /\ Col I C D)



tarski_s_parallel_postulate :=
 ∀ A B C D T : Tpoint,
 Bet A D T -> Bet B D C -> A<>D ->
 ∃ x, ∃ y,
 Bet A B x /\ Bet A C y /\ Bet x T y


This unicity of parallel postulate.


playfair_s_postulate :=
 ∀ A1 A2 B1 B2 C1 C2 P,
 Par A1 A2 B1 B2 -> Col P B1 B2 ->
 Par A1 A2 C1 C2 -> Col P C1 C2 ->
 Col C1 B1 B2 /\ Col C2 B1 B2


According to wikipedia: "Proclus (410-485) wrote a commentary on The Elements where he comments on attempted proofs to deduce the fifth postulate from the other four, in particular he notes that Ptolemy had produced a false 'proof'. Proclus then goes on to give a false proof of his own. However he did give a postulate which is equivalent to the fifth postulate."


proclus_postulate :=
 ∀ A B C D P Q, Par A B C D -> Col A B P -> ~ Col A B Q -> ∃ Y, Col P Q Y /\ Col C D Y


Transitivity of parallelism.


postulate_of_transitivity_of_parallelism :=
 ∀ A1 A2 B1 B2 C1 C2,
 Par A1 A2 B1 B2 -> Par B1 B2 C1 C2 -> Par A1 A2 C1 C2


perpendicular_transversal_postulate :=
 ∀ A B C D P Q,
 Par A B C D -> Perp A B P Q -> Perp C D P Q


postulate_of_parallelism_of_perpendicular_tranversals :=
 ∀ A1 A2 B1 B2 C1 C2 D1 D2,
 Par A1 A2 B1 B2 -> Perp A1 A2 C1 C2 -> Perp B1 B2 D1 D2 -> Par C1 C2 D1 D2


triangle_circumscription_principle :=
  ∀ A B C,
  ~ Col A B C -> ∃ CC, Cong A CC B CC /\ Cong A CC C CC


BetS (A B C : Tpoint) : Prop := Bet A B C /\ A <> B /\ B <> C


Lemma BetSEq : ∀ A B C, BetS A B C <-> Bet A B C /\ A <> B /\ A <> C /\ B <> C.
Proof.
intros; unfold BetS; split; intro; spliter;
repeat split; Col; intro; treat_equalities; Col.
Qed.


SPP :=
  ∀ P Q R S T U,
  BetS P T Q -> BetS R T S -> ~ Col P R U ->
  coplanar P Q R U ->
  Cong P T Q T -> Cong R T S T ->
  ∃ I, Col S Q I /\ Col P U I


strong_parallel_postulate :=
  ∀ P Q R S T U,
  BetS P T Q -> BetS R T S -> ~ Col P R U ->
  Cong P T Q T -> Cong R T S T ->
  ∃ I, Col S Q I /\ Col P U I


euclid_5 :=
  ∀ P Q R S T U,
  BetS P T Q -> BetS R T S -> BetS Q U R -> ~ Col P Q S ->
  Cong P T Q T -> Cong R T S T ->
  ∃ I, BetS S Q I /\ BetS P U I


midpoints_converse_postulate :=
  ∀ A B C P Q,
 ~Col A B C ->
 is_midpoint P B C ->
 Par A B Q P ->
 Col A C Q ->
 is_midpoint Q A C


End Euclid_def.
