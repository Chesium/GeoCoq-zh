Class incidence_geometry (Point:Type) (Line:Type) := {
 EqL : Line → Line → Prop;
 EqL_Equiv : Equivalence EqL;
 Incid : Point → Line → Prop;
 line_existence : ∀ A B, A≠B → ∃ l, Incid A l ∧ Incid B l;
 line_unicity : ∀ A B l m, A ≠ B → Incid A l → Incid B l → Incid A m → Incid B m → EqL l m;
 two_points_on_line : ∀ l, ∃ A, ∃ B, Incid B l ∧ Incid A l ∧ A ≠ B;
 ColH := fun A B C ⇒ ∃ l, Incid A l ∧ Incid B l ∧ Incid C l;
 plan : ∃ A, ∃ B, ∃ C, ¬ ColH A B C
}.

Class distance_function (Point:Type) := {
 d : Point → Point → R;
 d_pos : ∀ P Q, d P Q > 0;
 d_identity : ∀ P Q, d P Q = 0 ↔ P=Q;
 d_sym : ∀ P Q, d P Q = d Q P
}.

Section bijections.

Variables A B : Type.
Definition injective (f : A → B) := ∀ x y : A, f x = f y → x = y.
Definition surjective (f : A → B) := ∀ y : B, ∃ x : A, f x = y.
Definition bijective (f : A → B) : Prop := injective f ∧ surjective f.

End bijections.

Definition is_rule (f:Point → R) l :=
 bijective Point R f ∧
 (∀ P Q : Point, Incid P l → Incid Q l → Rabs (f P - f Q) = d P Q).

Class metric_geometry := {
 ruler_postulate : ∀ l : Line, ∃ f, is_rule f l
}.

End S1.

Definition between A B C :=
 ColH A B C ∧
 d A B + d B C = d A C.

Lemma d_zero : ∀ A, d A A = 0.
Proof.
intros.
apply d_identity.
reflexivity.
Qed.

Lemma between_identity : ∀ A B, between A B A → A = B.
Proof.
intros.
unfold between in ×.
destruct H.
replace (d B A) with (d A B) in × by apply d_sym.
rewrite d_zero in ×.
replace (d A B + d A B) with (2×d A B) in × by ring.
apply d_identity.

apply Rmult_integral in H0.
elim H0.
intro.
assert (2≠0).
discrR.
intuition.
auto.
Qed.

Definition cong A B C D :=
  d A B = d C D.

Lemma cong_identity : ∀ A B C, cong A B C C → A = B.
Proof.
intros.
unfold cong in ×.
rewrite d_zero in H.
apply d_identity.
assumption.
Qed.

Lemma cong_pseudo_reflexivity : ∀ A B, cong A B B A.
Proof.
apply d_sym.
Qed.

Lemma cong_inner_transitivity : ∀ A B C D E F,
 cong A B C D → cong A B E F → cong C D E F.
Proof.
unfold cong;intros.
rewrite <- H.
assumption.
Qed.
