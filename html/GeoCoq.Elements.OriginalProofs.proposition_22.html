Lemma lemma_togethera
     : ∀ A B C D F G P Q a b c : Point,
       TG A a B b C c →
       Cong D F A a →
       Cong F G B b →
       BetS D F G →
       Cong P Q C c → Lt P Q D G.
Proof.
intros.
apply (lemma_together A B C D F G P Q a b c);auto.
Qed.
