Definition Bet A B C := BetS A B C ∨ A=B ∨ B=C.

Lemma Bet_sym : ∀ A B C,
  Bet A B C → Bet C B A.
Proof.
intros.
unfold Bet in ×.
decompose [or] H.
left.
eauto using axiom_betweennesssymmetry.
right;right;auto.
right;left;auto.
Qed.

 unfold Bet in ×.
