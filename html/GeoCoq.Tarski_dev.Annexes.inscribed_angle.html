Lemma inscribed_angleg : ∀ O P A B C,
  O ≠ P → OS A B O C → OnCircle A O P → OnCircle B O P → OnCircle C O P →
  gHalfA A C B A O B.
Proof.
intros.
unfold gHalfA.
assert(HH:=inscribed_angleg_aux O P A B C H H0 H1 H2 H3).
ex_and HH T.
∃ T.
split; CongA.
unfold HalfA.
split.
intro.
unfold OS in H0.
ex_and H0 R.
unfold TS in H0.
spliter.
apply H0.
Col.
split; auto.
apply (conga_trans _ _ _ A C B); CongA.
Qed.

End Inscribed_angle.
