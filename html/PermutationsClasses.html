Fixpoint mapi_aux {A B:Type} (c:nat) (f:nat -> A -> B) (l: list A) :=
 match l with
  | nil => nil
  | a::r => f c a::mapi_aux (S c) f r
end.

Lemma length_mapi_aux_aux {A B:Type} :
  forall (f:nat -> A -> B) (l: list A) (m n:nat),
  length (mapi_aux m f l) = length (mapi_aux n f l).
Proof.
intros; revert m; revert n.
induction l; simpl; auto.
Defined.

Lemma length_mapi_aux {A B:Type} :
  forall (f:nat -> A -> B) (l: list A) (m:nat),
  length (mapi_aux m f l) = length l.
Proof.
intros.
induction l; simpl; auto.
rewrite <- IHl; apply eq_S.
rewrite (length_mapi_aux_aux f l (S m) m); auto.
Defined.

Definition mapi {A B:Type} (f:nat -> A -> B) (l: list A) := mapi_aux 1 f l.

Definition apply_perm {T:Type} (A:T) (l:list T) (m:list nat) :=
  mapi (fun i x => (nth (nth i m 0) l A)) m.

Lemma apply_perm_length : forall (T:Type) (A:T) l m,
  length (apply_perm A l m) = length m.
Proof.
intros.
unfold apply_perm, mapi; rewrite length_mapi_aux; auto.
Defined.

Definition apply_perm_CP {Ty: Type} (A:Ty) (n:nat) (Ps: cartesianPower Ty n) (s: cartesianPower nat n) : (cartesianPower Ty n).
Proof.
assert (X:=
 ListToCP (apply_perm A (CPToList Ps) (CPToList s)) A).
rewrite apply_perm_length in X.
rewrite <- lengthOfCPToList in X.
exact X.
Defined.

Definition app_partial (Ty:Type) (q:nat) (P: cartesianPower Ty (S q) -> Prop) (x: Ty) : (cartesianPower Ty q) -> Prop.
Admitted.

Program Fixpoint put_forall (Ty:Type) (q:nat) (P: cartesianPower Ty q -> Prop) :=
 match q with
   0 => True
| S p => forall x:Ty, put_forall Ty p (app_partial Ty p P x)
end.

Class Perm {Ty: Type} (A:Ty) (n:nat) (P: arity Ty n) (perms: list (cartesianPower nat n)) :=
{
  permutations_prop :
  put_forall2 Ty n (forall Ps : (cartesianPower Ty n), (fold_left or (map (fun s => arity.app P (apply_perm_CP A n Ps s)) perms) False) -> arity.app P Ps)
}.

Class Perm4LR {Ty:Type} (P: Ty -> Ty -> Ty -> Ty -> Prop) :=
 {
   left_comm : forall A B C D, P A B C D -> P B A C D;
   right_comm : forall A B C D, P A B C D -> P A B D C
}.


Class Perm4LRC {Ty:Type} (P: Ty -> Ty -> Ty -> Ty -> Prop) :=
 {
  perm_lr : Perm4LR P;
  swap : forall A B C D, P A B C D -> P C D A B
}.


Class Perm4C {Ty:Type} (P: Ty -> Ty -> Ty -> Ty -> Prop) :=
 {
   circular : forall A B C D, P A B C D -> P B C D A;
   lr_comm : forall A B C D, P A B C D -> P B A D C
}.


Class Perm3A {Ty:Type} (P: Ty -> Ty -> Ty -> Prop) :=
 {
   circular3 : forall A B C, P A B C -> P B C A;
   swap3 : forall A B C, P A B C -> P B A C
}.


Class Perm4A {Ty:Type} (P: Ty -> Ty -> Ty -> Ty -> Prop) :=
 {
   circular4 : forall A B C D , P A B C D -> P B C D A;
   swap4 : forall A B C D, P A B C D -> P B A C D
}.


Class Perm3S {Ty:Type} (P: Ty -> Ty -> Ty -> Prop) :=
 {
   sym : forall A B C, P A B C -> P C B A
}.


Class Perm6A {Ty:Type} (P: Ty -> Ty -> Ty -> Ty -> Ty -> Ty -> Prop) :=
 {
    left_comm_6 : forall A B C D E F, P A B C D E F -> P C B A D E F;
    right_comm_6 : forall A B C D E F, P A B C D E F -> P A B C F E D
}.
