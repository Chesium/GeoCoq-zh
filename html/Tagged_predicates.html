Diff_tagged (A B: Tpoint) := A <> B


Lemma Diff_Diff_tagged : ∀ A B , A <> B -> Diff_tagged A B.
Proof.
trivial.
Qed.


Lemma Diff_tagged_Diff : ∀ A B , Diff_tagged A B -> A <> B.
Proof.
trivial.
Qed.


Lemma Diff_perm :
  ∀ (A B: Tpoint),
  A <> B ->
  A <> B /\ B <> A.
Proof.
intros.
repeat split; intuition.
Qed.
