Library lombard_vesley_axioms


We list here the axioms from: "A common axiom set for classical and intuitionistic plane geometry", Melinda Lombard and Richard Vesley, Annals of Pure and Applied Logic 95 (1998), 229-255.

Class EGA := {
 Tpoint : Type;
 Dist : Tpoint -> Tpoint -> Tpoint ->
        Tpoint -> Tpoint -> Tpoint -> Prop;
 EGA1: forall A B C D E F G, Dist A B C D E F -> Dist A B C D G G;
 EGA2: forall A B C D E F, Dist A B C D E F -> Dist A B D D E F;
 EGA3: forall A B C D E F, Dist A B C D E F -> Dist B A C D E F /\ Dist A B C D F E;
 EGA4: forall A B C D E F, Dist A B C D E F -> Dist C D A B E F;
 Dist4 A B C D := Dist A B B B C D;
 EGA5: forall A B C D, Dist4 A B C D -> ~ Dist4 A B C D;
 EGA6: forall A B C D E F X Y, Dist A B C D E F -> ~ Dist4 X Y E F -> Dist A B C D X Y;
 EGA7: forall A B C D E F X Y, Dist A B C D E F -> ~ Dist4 A B X Y -> Dist X Y C D E F;
 EGA8: forall A B C D E F, Dist4 A B C D -> Dist4 A B E F \/ Dist4 E F C D;
 Apt A B := Dist4 A B B B;
 Equ A B := ~ Apt A B;
 EGA9_1: forall A B C D E F X, Dist A B C D E F -> Apt A X \/ Dist X B C D E F;
 EGA9_2: forall A B C D E F X, Dist A B C D E F -> Apt B X \/ Dist A X C D E F;
 EGA9_3: forall A B C D E F X, Dist A B C D E F -> Apt C X \/ Dist A B X D E F;
 EGA9_4: forall A B C D E F X, Dist A B C D E F -> Apt D X \/ Dist A B C X E F;
 EGA9_5: forall A B C D E F X, Dist A B C D E F -> Apt E X \/ Dist A B C D X F;
 EGA9_6: forall A B C D E F X, Dist A B C D E F -> Apt F X \/ Dist A B C D E X;
 EGA10 : forall A B C D E F, Dist A B C D E F /\ Dist C D E F A B -> Apt C D
}.

Class EGB (T: EGA) := {
 DC A B C D:= ~ Dist4 A B C D /\ ~ Dist4 C D A B;
 TR a b c := Dist a b b c a c /\ Dist a c b c a b /\ Dist a c a b b c;
 B a b c := ~ Dist a b b c a c;
 Bet a b c := B a b c /\ Apt a b /\ Apt b c;
 EGB1 : forall a b c d, B a b c /\ B a c d -> B b c d;
 EGB2 : forall a b c d, B a b d /\ B b c d -> B a c d;
 EGB3 : forall a b c d, Apt b c -> B a b c -> B b c d -> B a b d;
 EGB4 : forall a b c, B a b c -> Apt b c -> Dist4 a c a b;
 EGB5 : forall a b c d, B a b c -> B a b d -> Apt a b -> Apt c d -> B a c d \/ B a d c;
 EGB6_1 : forall a b c d e, B a b c -> Dist a b b c d e -> Dist4 a c d e;
 EGB6_2 : forall a b c d e, B a b c -> Dist4 a c d e -> Dist a b b c d e ;
 EGB7 : forall a b c x, TR a b c -> Apt c x \/ TR a b x
}.

Section EGA.

Context `{M:EGA}.

Hint Resolve EGA1 EGA2 EGA3 EGA4 EGA5 EGA6 EGA7 EGA8
             EGA9_1 EGA9_2 EGA9_3 EGA9_4 EGA9_5 EGA9_6
             EGA10 : EGA.

Hint Unfold Dist4 Apt Equ : EGA.

Lemma EG1 : forall a b c d e f,
 Dist4 a b c d -> Dist4 c d e f -> Dist4 a b e f.
Proof.
intros.
unfold Dist4 in *.
eauto 7 using EGA.
intuition (firstorder using EGA).
Admitted.

Lemma EG2 : forall a b c,
 Dist4 a b a c -> Apt b c.
Proof.
intros.
Admitted.

Lemma EG3_1 : forall a,
 Equ a a.
Proof.
intros.
unfold Equ.
unfold Apt.
intro.
assert (T:= EGA5 a a a a H).
apply T.
apply H.
Qed.

Lemma EG3_2 : forall a b,
 Equ a b -> Equ b a.
Proof.
intros.
unfold Equ, Apt in *.
unfold Dist4 in *.

Admitted.

Lemma EG3_3 : forall a b c,
 Equ a b -> Equ b c -> Equ a c.
Proof.
intros.
Admitted.

Lemma EG5 : forall a b,
 ~ ~ Equ a b -> Equ a b.
Proof.
intros.
Admitted.

Lemma EG6 : forall a b,
 Apt a b -> Apt b a.
Proof.
intros.
Admitted.

Lemma EG7 : forall a b c,
 Apt a b -> Apt c a \/ Apt c b.
Proof.
intros.
Admitted.

End EGA.

Admitted.

Admitted.

Lemma EG20: forall a b c d e f u w y z,
 DC y z a b -> DC z u c d -> DC y w e f-> Dist a b c d e f -> Dist y z z u y w.

Lemma EG21_1: forall a b c d e f x y z u w,
 Apt x y -> B x y z -> B x z u -> B x y w ->
 DC y z a b -> DC z u c d -> DC y w e f ->
 Dist a b c d e f -> Bet x w u.

Admitted.

Lemma EG21_2: forall a b c d e f x y z u w,

 DC y z a b -> DC z u c d -> DC y w e f ->
 Bet x w u -> Dist a b c d e f.

Admitted.

Definition Col a b c := exists d, exists e, Apt d e /\ Bet d e a /\ Bet d e b /\ Bet d e c.

Lemma EG22: forall a b c,
 Col a b c -> Apt a b -> Apt b c -> Apt a c -> Bet a b c \/ Bet b c a \/ Bet c a b.
Proof.
Admitted.

Lemma EG23: forall a b c,
 Col a b c -> B a b c \/ B b c a \/ B c a b.
Proof.
Admitted.

Lemma EG24: forall a b c d,
 B a b c -> B a b d -> ~ Equ a b -> B a c d \/ B a d c.
Proof.
Admitted.

End EGB.
