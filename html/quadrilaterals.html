Lemma Rhombus_Rectangle_Square : ∀ A B C D,
 Rhombus A B C D ->
 Rectangle A B C D ->
 Square A B C D.
Proof.
intros.
unfold Square.
unfold Rhombus in *.
tauto.
Qed.


Lemma rhombus_cong_square : ∀ A B C D,
 Rhombus A B C D ->
 Cong A C B D ->
 Square A B C D.

intros.
apply Rhombus_Rectangle_Square.
assumption.
apply Rhombus_Plg in H.
apply plg_cong_rectangle;auto.
Qed.
